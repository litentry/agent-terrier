# ESP-IDF component: build + link the shared `agentkeys-device-core` Rust crate as
# a no_std staticlib for Xtensa (issue #367 anti-drift). The K10 crypto the device
# runs (secp256k1 keygen, device_key_hash, EIP-191 pop_sig) is the SAME source the
# daemon + broker use — compiled here, NOT reimplemented in C.
#
# Prerequisite: the esp-rs toolchain (Xtensa needs Espressif's LLVM fork). Install
# it but do NOT source ~/export-esp.sh into the IDF build — that env prepends espup's
# xtensa-esp-elf-gcc and SHADOWS ESP-IDF's own GCC (its tool_version_check then fails).
# We select the toolchain per-invocation with `cargo +esp`, which needs no GCC for a
# pure-Rust archive. setup-esp32.sh does this for you; by hand:
#   cargo install espup && espup install && rustup component add rust-src --toolchain esp
# The header in include/ is generated by ../../../../crates/agentkeys-device-core/gen-header.sh.
idf_component_register(INCLUDE_DIRS "include")

# Repo root: components/agentkeys_device -> esp32s3-touch-lcd-4b -> firmware -> <root>
get_filename_component(_ak_repo_root "${CMAKE_CURRENT_LIST_DIR}/../../../.." ABSOLUTE)
set(_ak_target "xtensa-esp32s3-none-elf")
set(_ak_out "${CMAKE_CURRENT_BINARY_DIR}/cargo")
set(_ak_lib "${_ak_out}/${_ak_target}/release/libagentkeys_device_core.a")

# Build the staticlib with the esp toolchain. `-Zbuild-std=core,alloc` compiles
# core+alloc from rust-src — the esp toolchain ships rust-src, NOT a precompiled std
# for xtensa-esp32s3-none-elf, so without it rustc errors E0463 "can't find crate for
# core". panic=abort (no unwinder on device); an isolated CARGO_TARGET_DIR keeps this
# cross build out of the host workspace target/. `cargo rustc -p <crate>` builds only
# this crate + its deps, so the rest of the (std) workspace is never compiled for Xtensa.
add_custom_command(
    OUTPUT "${_ak_lib}"
    COMMAND ${CMAKE_COMMAND} -E env "CARGO_TARGET_DIR=${_ak_out}"
            cargo +esp rustc
                --manifest-path "${_ak_repo_root}/Cargo.toml"
                --package agentkeys-device-core
                --release
                --features freestanding
                --target ${_ak_target}
                --crate-type staticlib
                -Zbuild-std=core,alloc
                -- -Cpanic=abort
    WORKING_DIRECTORY "${_ak_repo_root}"
    COMMENT "cargo: agentkeys-device-core (no_std) -> ${_ak_target}"
    VERBATIM
)
add_custom_target(agentkeys_device_cargo DEPENDS "${_ak_lib}")

add_prebuilt_library(agentkeys_device_rs "${_ak_lib}")
add_dependencies(agentkeys_device_rs agentkeys_device_cargo)

# Anything that depends on this component (main) links the staticlib + sees include/.
target_link_libraries(${COMPONENT_LIB} INTERFACE agentkeys_device_rs)
